Insertion at the Kth position of Linked List

Easy dsa
Linked List

Problem Statement

Given the head of a singly linked list and two integers X and K, insert a node with value X as the kth node of the linked list and return the head of the modified list.

Constraints

  • n == number of nodes in the Linked List
  • 0 <= n <= 1000
  • 0 <= ListNode.val <= 100
  • 0 <= X <= 100
  • 1 <= K <= n+1

Example 1

Input: linkedList = [1, 2, 3], X = 5, K = 2

Output: [1, 5, 2, 3]

Example 2

Input: linkedList = [], X = 7, K = 1

Output: [7]

Explanation:

Note that the value of the head was changed.

Example 3

Input: linkedList = [1, 2], X = 15, K = 3

Editorial: Solution

Intuition:

The intuition is to traverse the linked list up to the position just before the desired insertion point. Once there, create a new node with the given value and update the links so that the new node is inserted at the correct position. This involves pointing the new node to the subsequent node and updating the previous node to point to the new node.

Approach:

  • Initialize a temporary pointer to traverse the list and a counter variable to track the number of nodes traversed.
  • Move the pointer until you reach the node just before the desired insertion position (k-1).
  • Create the new node with the given value.
  • To insert the new node:
    • Set the new node's next pointer to the current (k)th node (temp->next).
    • Update the (k-1)th node's next pointer to the new node, completing the insertion.
  • If the traversal finishes before reaching the (k-1)th node, it means k is larger than the length of the list plus one, so no node will be inserted.

// Definition of singly linked list
struct ListNode
{
    int val;
    ListNode *next;
    ListNode(int data1)
    {
        val = data1;
        next = NULL;
    }
    ListNode(int data1, ListNode *next1)
    {
        val = data1;
        next = next1;
    }
};

// Solution class
class Solution {
public:
    // Function to insert a new node at the kth position 
    ListNode* insertAtKthPosition(ListNode* &head, int X, int K) {
       /* If the linked list is empty 
       and k is 1, insert the 
       new node as the head*/
        if (head == NULL) {
            if (K == 1)
                return new ListNode(X);
            else
                return head;
        }
    
        /*If K is 1, insert the new
         node at the beginning 
         of the linked list*/
        if (K == 1)
            return new ListNode(X, head);
    
        int cnt = 0;
        ListNode* temp = head;
    
       /* Traverse the linked list 
       to find the node at position k-1*/
        while (temp != NULL) {
            cnt++;
            if (cnt == K-1) {
               /* Insert the new node after the node 
               at position k-1*/
                ListNode* newNode = new ListNode(X, temp->next);
                temp->next = newNode;
                break;
            }
            temp = temp->next;
        }
    
        return head;
    }
};

        cout << head->val << " ";
        head = head->next;
    }
    cout << endl;
}

package main

import "fmt"

// ListNode defines a node in the singly linked list
type ListNode struct {
    Data  int
    Next *ListNode
}

// insertAtKthPosition inserts a new node with value X at position K (1-indexed)
func insertAtKthPosition(head *ListNode, X int, K int) *ListNode {
    /* If the linked list is empty 
       and k is 1, insert the 
       new node as the head */
    if head == nil {
        if K == 1 {
            return &ListNode{Data: X}
        }
        return head
    }

    /* If K is 1, insert the new
       node at the beginning 
       of the linked list */
    if K == 1 {
        return &ListNode{Data: X, Next: head}
    }

    cnt := 0
    temp := head

    /* Traverse the linked list 
       to find the node at position k-1 */
    for temp != nil {
        cnt++
        if cnt == K-1 {
            /* Insert the new node after the node 
               at position k-1 */
            newNode := &ListNode{Data: X, Next: temp.Next}
            temp.Next = newNode
            break
        }
        temp = temp.Next
    }

    return head
}

// printLL prints the linked list values
func printLL(head *ListNode) {
    for head != nil {
        fmt.Printf("%d ", head.Data)
        head = head.Next
    }
    fmt.Println()
}

func main() {
    // Create a linked list from a slice: 10 -> 30 -> 40
    arr := []int{10, 30, 40}
    var head *ListNode
    for i := len(arr) - 1; i >= 0; i-- {
        head = &ListNode{Data: arr[i], Next: head}
    }

    X, K := 20, 2

    // Print the original list
    fmt.Print("Original List: ")
    printLL(head)

    // Insert value X at the K-th position
    head = insertAtKthPosition(head, X, K)

    // Print the modified linked list
    fmt.Print("List after inserting the given value at the Kth position: ")
    printLL(head)
}
