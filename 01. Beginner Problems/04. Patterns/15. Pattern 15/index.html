Editorial: Solution

Approach: 

  • Use a for loop to iterate from 0 to N-1, where N is the number of rows. This loop will ensure to print each row of the pattern.
  • The inner loop will run for i times and print alphabets from 'A' to 'A' + (N-row number-1).
  • After completing a row give a line break, to make sure next row gets printed as well.

    class Solution {
    // Function to print pattern15
    void pattern15(int n) {
        // Outer loop for the number of rows.
        for (int i = 0; i < n; i++) {
            
            /*Inner loop will loop for i times and
            print alphabets from A to A + (n - i - 1).*/
            for (char ch = 'A'; ch<='A'+(n-i-1);ch++) {
                System.out.print(ch);
            }
            
            /*As soon as the letters for each iteration
            are printed, we move to the next row and give
            a line break otherwise all letters would get
            printed in 1 line.*/
            System.out.println();
        }
    }

    public static void main(String[] args) {
        int N = 5;

        // Create an instance of Solution class
        Solution sol = new Solution();

        sol.pattern15(N);
    }
}

    class Solution {
    // Function to print pattern15
    pattern15(n) {
        // Outer loop for the number of rows.
        for (let i = 0; i < n; i++) {
            
            /* Initialize an empty string to 
            accumulate characters for the current row*/
            let row = "";

            /* Inner loop will loop for (n - i) times and
            print alphabets from 'A' to 'A' + i.*/
            for (let j = 0, ch = 'A'; j <= (n - i - 1); j++, ch = String.fromCharCode(ch.charCodeAt(0) + 1)) {
                row += ch;
            }
            
            /* Print the accumulated row string
            for the current iteration*/
            console.log(row);
        }
    }
}

    public void pattern15(int n)
    {
        // Outer loop for the number of rows.
        for (int i = 0; i < n; i++)
        {
            
            /* Initialize a StringBuilder to 
            accumulate characters for the current row*/
            StringBuilder rowBuilder = new StringBuilder();

            /* Inner loop will loop for (n - i) times and

            for (int j = 0; j <= (n - i - 1); j++)
            {
                rowBuilder.Append((char)('A' + j));
            }
            
            /* Print the accumulated row string
            for the current iteration*/
            Console.WriteLine(rowBuilder.ToString());
        }
    }

    package main

import "fmt"

// Function to print pattern15
func pattern15(n int) {
	// Outer loop for the number of rows.
	for i := 0; i < n; i++ {

		/*Inner loop will loop for i times and
		print alphabets from A to A + (n-i-1).*/
		for ch := 'A'; ch <= 'A'+rune(n-i-1); ch++ {
			fmt.Print(string(ch))
		}

		/* As soon as the letters for each iteration
		are printed, we move to the next row and give
		a line break otherwise all letters would get
		printed in 1 line.*/
		fmt.Println()
	}
}

func main() {
	N := 5

	// Function call to print pattern15
	pattern15(N)
}
